250// 2-D Constructor
251Gradient::Gradient(u16 k, u32 m, u32 n, Real dx, Real dy) {
252 Gradient Gx(k, m, dx);
253 Gradient Gy(k, n, dy);
254
255 sp_mat Im = speye(m + 2, m + 2);
256 sp_mat In = speye(n + 2, n + 2);
257
258 Im.shed_row(0);
259 Im.shed_row(m);
260 In.shed_row(0);
261 In.shed_row(n);
262
263 sp_mat G1 = Utils::spkron(In, Gx);
264 sp_mat G2 = Utils::spkron(Gy, Im);
265
266 // Dimensions = 2*m*n+m+n, (m+2)*(n+2)
267 if (m != n)
268 *this = Utils::spjoin_cols(G1, G2);
269 else {
270 sp_mat A1(2, 1);
271 sp_mat A2(2, 1);
272 A1(0, 0) = A2(1, 0) = 1.0;
273 *this = Utils::spkron(A1, G1) + Utils::spkron(A2, G2);
274 }
275}

277// 3-D Constructor
278Gradient::Gradient(u16 k, u32 m, u32 n, u32 o, Real dx, Real dy, Real dz) {
279 Gradient Gx(k, m, dx);
280 Gradient Gy(k, n, dy);
281 Gradient Gz(k, o, dz);
282
283 sp_mat Im = speye(m + 2, m + 2);
284 sp_mat In = speye(n + 2, n + 2);
285 sp_mat Io = speye(o + 2, o + 2);
286
287 Im.shed_row(0);
288 Im.shed_row(m);
289 In.shed_row(0);
290 In.shed_row(n);
291 Io.shed_row(0);
292 Io.shed_row(o);
293
294 sp_mat G1 = Utils::spkron(Utils::spkron(Io, In), Gx);
295 sp_mat G2 = Utils::spkron(Utils::spkron(Io, Gy), Im);
296 sp_mat G3 = Utils::spkron(Utils::spkron(Gz, In), Im);
297
298 // Dimensions = 3*m*n*o+m*n+m*o+n*o, (m+2)*(n+2)*(o+2)
299 if ((m != n) || (n != o))
300 *this = Utils::spjoin_cols(Utils::spjoin_cols(G1, G2), G3);
301 else {
302 sp_mat A1(3, 1);
303 sp_mat A2(3, 1);
304 sp_mat A3(3, 1);
305 A1(0, 0) = A2(1, 0) = A3(2, 0) = 1.0;
306 *this =
307 Utils::spkron(A1, G1) + Utils::spkron(A2, G2) + Utils::spkron(A3, G3);
308 }
309}

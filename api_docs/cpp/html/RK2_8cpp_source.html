3// Description:
4// This program solves a first-order ordinary differential equation (ODE) of the form:
5// dy/dt = sin^2(t) * y
6// using the second-order Runge-Kutta (RK2) method. The solution is computed over the
7// time interval [0, 5] with an initial condition y(0) = 2.0.
8
9#include "mole.h"

11#include <sstream> // For string streams

14#include <iomanip> // For setprecision

16// Function prototype for f(t, y)
17double f(double t, double y);
18
19int main() {
20 constexpr double h = 0.1; // Step size
21 constexpr double t_start = 0.0; // Initial time
22 constexpr double t_end = 5.0; // Final time
23
24 int n_steps = static_cast<int>((t_end - t_start) / h) + 1;
25
26 // MOLE's vec type for vectors
27 vec t(n_steps); // Time vector
28 vec y(n_steps); // Solution vector
29
30 // Initial conditions
31 t(0) = t_start;
32 y(0) = 2.0;
33
34 // Populate the time vector
35 for (int i = 1; i < n_steps; ++i) {
36 t(i) = t(i - 1) + h;
37 }
38
39 // RK2 Method
40 for (int i = 0; i < n_steps - 1; ++i) {
41 const double k1 = f(t(i), y(i)); // Slope at the beginning
42 const double k2 = f(t(i) + h / 2.0, y(i) + h / 2.0 * k1); // Slope at midpoint
43 y(i + 1) = y(i) + h * k2; // Update solution
44 }
45
46 // Set the output stream to fixed-point notation with 6 decimal places
47 std::cout << std::fixed << std::setprecision(6);

0017 function [J, Xe, Xn, Xc, Ye, Yn, Yc, Ze, Zn, Zc] = jacobian3D(k, X, Y, Z)
0018 % Returns:
0019 %                J : Determinant of the Jacobian
0020 %               Xe : dx/de metric
0021 %               Xn : dx/dn metric
0022 %               Xc : dx/dc metric
0023 %               Ye : dy/de metric
0024 %               Yn : dy/dn metric
0025 %               Yc : dy/dc metric
0026 %               Ze : dz/de metric
0027 %               Zn : dz/dn metric
0028 %               Zc : dz/dc metric
0029 %
0030 % Parameters:
0031 %                k : Order of accuracy
0032 %                X : x-coordinates (physical) of meshgrid
0033 %                Y : y-coordinates (physical) of meshgrid
0034 %                Z : z-coordinates (physical) of meshgrid
0035     
0036     [n, m, o] = size(X);
0037     
0038     X = reshape(permute(X, [2, 1, 3]), [], 1);
0039     Y = reshape(permute(Y, [2, 1, 3]), [], 1);
0040     Z = reshape(permute(Z, [2, 1, 3]), [], 1);
0041     
0042     N = nodal3D(k, m, 1, n, 1, o, 1);
0043     
0044     X = N*X;
0045     Y = N*Y;
0046     Z = N*Z;
0047     
0048     mno = m*n*o;
0049     
0050     Xe = X(1:mno);
0051     Xn = X(mno+1:2*mno);
0052     Xc = X(2*mno+1:end);
0053     Ye = Y(1:mno);
0054     Yn = Y(mno+1:2*mno);
0055     Yc = Y(2*mno+1:end);
0056     Ze = Z(1:mno);
0057     Zn = Z(mno+1:2*mno);
0058     Zc = Z(2*mno+1:end);
0059     
0060     J = Xe.*(Yn.*Zc-Yc.*Zn)-Ye.*(Xn.*Zc-Xc.*Zn)+Ze.*(Xn.*Yc-Xc.*Yn);
0061 end

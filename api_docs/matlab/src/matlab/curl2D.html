0017 function C = curl2D(k, m, dx, n, dy, west, east, south, north, U, V)
0018 % Returns a two-dimensional mimetic curl operator
0019 %
0020 % Parameters:
0021 %                k : Order of accuracy
0022 %                m : Number of cells along x-axis
0023 %               dx : Step size along x-axis
0024 %                n : Number of cells along y-axis
0025 %               dy : Step size along y-axis
0026 %      west, east,
0027 %     south, north : west, east, south, north limits
0028 %                U : Vector space function acting on x-direction
0029 %                    U(X,Y) must be defined as function handle
0030 %                V : Vector space function acting on y-direction
0031 %                    V(X,Y) must be defined as function handle
0032 
0033 
0034 % Arguments validation
0035     assert(isa(U, 'function_handle') && isa(V, 'function_handle'), ...
0036            'U and V must be defined as function handle');
0037     assert(nargin(U) == 2 && nargin(V) == 2, ...
0038            'U and V must be defined as 2D vector functions');
0039     assert(west < east && south < north , 'west < east, south < north');
0040 
0041 % Variables initialization
0042     F = sparse(2*m*n+m+n, 1);
0043     xaxis = west : (dx/2) : east;
0044     yaxis = south : (dy/2) : north;
0045 
0046 % Linearizing F as function of U and V: F = V_x - U_y
0047     f = 1;
0048     for j = 2 : 2 : 2*n+1
0049         for i = 1 : 2 : 2*m+1
0050             F(f) = V(xaxis(i), yaxis(j));  % F in the x direction (V_x)
0051             f = f + 1;
0052         end
0053     end
0054 
0055     for j = 1 : 2 : 2*n+1
0056         for i = 2 : 2 : 2*m+1
0057             F(f) = -U(xaxis(i), yaxis(j));  % F in the y direction (-U_y)
0058             f = f + 1;
0059         end
0060     end
0061 
0062     C = div2D(k, m, dx, n, dy)*F;  % Mimetic curl as C = Div2D*F
0063     C = reshape(C, m+2, n+2);      % Reshaping C as matrix
0064     C = C(2:end-1, 2:end-1);       % Removing boundaries
0065 end

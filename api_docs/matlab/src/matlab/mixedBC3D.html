0017 function BC = mixedBC3D(k, m, dx, n, dy, o, dz, left, coeffs_left, right, coeffs_right, bottom, coeffs_bottom, top, coeffs_top, front, coeffs_front, back, coeffs_back)
0018 % Constructs a 3D mimetic mixed boundary conditions operator
0019 %
0020 % Parameters:
0021 %    k               : Order of accuracy
0022 %    m               : Number of cells in x-direction
0023 %    dx              : Step size in x-direction
0024 %    n               : Number of cells in y-direction
0025 %    dy              : Step size in y-direction
0026 %    o               : Number of cells in z-direction
0027 %    dz              : Step size in z-direction
0028 %    left            : Type of boundary condition at the left boundary ('Dirichlet', 'Neumann', 'Robin')
0029 %    coeffs_left     : Coefficients for the left boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0030 %    right           : Type of boundary condition at the right boundary ('Dirichlet', 'Neumann', 'Robin')
0031 %    coeffs_right    : Coefficients for the right boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0032 %    bottom          : Type of boundary condition at the bottom boundary ('Dirichlet', 'Neumann', 'Robin')
0033 %    coeffs_bottom   : Coefficients for the bottom boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0034 %    top             : Type of boundary condition at the top boundary ('Dirichlet', 'Neumann', 'Robin')
0035 %    coeffs_top      : Coefficients for the top boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0036 %    front           : Type of boundary condition at the front boundary ('Dirichlet', 'Neumann', 'Robin')
0037 %    coeffs_front    : Coefficients for the front boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0038 %    back            : Type of boundary condition at the back boundary ('Dirichlet', 'Neumann', 'Robin')
0039 %    coeffs_back     : Coefficients for the back boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0040 
0041     % 1-D boundary operators
0042     Bm = mixedBC(k, m, dx, left, coeffs_left, right, coeffs_right);
0043     Bn = mixedBC(k, n, dy, bottom, coeffs_bottom, top, coeffs_top);
0044     Bo = mixedBC(k, o, dz, front, coeffs_front, back, coeffs_back);
0045 
0046     Im = speye(m+2);
0047 
0048     In = speye(n+2);
0049 
0050     Io = speye(o+2);
0051     Io(1, 1) = 0;
0052     Io(end, end) = 0;
0053 
0054     In2 = In;
0055     In2(1, 1) = 0;
0056     In2(end, end) = 0;
0057 
0058     BC1 = kron(kron(Io, In2), Bm);
0059     BC2 = kron(kron(Io, Bn), Im);
0060     BC3 = kron(kron(Bo, In), Im);
0061 
0062     BC = BC1 + BC2 + BC3;
0063 end

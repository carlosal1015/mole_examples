0017 function I = DI2(m, n, type)
0018     if strcmp(type, 'Dn')
0019         e = ones(m, 1);
0020         bdry = spdiags([-0.5*e -0.5*e 0.5*e 0.5*e], [0 1 m+1 m+2], m, (m+1)*n);
0021         block = spdiags([[0.25*e; 0; 0] [0.25*e; 0.25; 0]], [0 1], m+2, m+1);
0022         pattern = spdiags([-ones(n-2, 1) ones(n-2, 1)], [0 2], n-2, n);
0023         middle = kron(pattern, block);
0024         I = [spalloc(m+3, (m+1)*n, 0); bdry; spalloc(2, (m+1)*n, 0); middle; circshift(bdry, (m+1)*(n-2), 2); spalloc(m+3, (m+1)*n, 0)];
0025     else
0026         e = ones(m-2, 1);
0027         block = spdiags([[-0.25*e; -0.25; 0] [0; 0.25*e; 0.25]], [-1 1], m+2, m);
0028         block(1, 1) = -0.5;
0029         block(1, 2) = 0.5;
0030         block(m, m-1) = -0.5;
0031         block(m, m) = 0.5;
0032         pattern = spdiags([ones(n, 1) ones(n, 1)], [0 1], n, n+1);
0033         middle = kron(pattern, block);
0034         I = [spalloc(m+3, (n+1)*m, 0); middle; spalloc(m+1, (n+1)*m, 0)];
0035     end
0036 end

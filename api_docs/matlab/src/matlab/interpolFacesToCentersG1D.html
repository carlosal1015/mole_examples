0017 function I = interpolFacesToCentersG1D(k, m)
0018 % 1D interpolation from faces to centers
0019 % centers logical coordinates [1,1.5:m-0.5,m]
0020 % m is the number of cells in the logical x-axis
0021 
0022     I = sparse(m+2, m+1);
0023 
0024     switch k
0025         case 2
0026             I(1, 1) = 2; I(end, end) = 2;
0027             for i = 2:m+1
0028                I(i, i-1:i) = [1 1];
0029             end
0030             denom = 2;
0031 
0032         case 4
0033             I(1, 1) = 128; I(end, end) = 128;
0034             A = [35 140 -70 28 -5];
0035             I(2, 1:5) = A;
0036             I(m+1, m-3:end) = fliplr(A);
0037             for i = 3:m
0038                I(i, i-2:i+1) = [-8 72 72 -8];
0039             end
0040             denom = 128;
0041 
0042         case 6
0043             I(1, 1) = 1024; I(end, end) = 1024;
0044             A = [231 1386 -1155  924 -495 154 -21; ... % (2,2) 1278
0045                  -21  378   945 -420  189 -54   7];
0046             I(2:3, 1:7) = A;
0047             I(m:m+1, m-5:end) = flipud(fliplr(A));%#ok
0048             for i = 4:m-1
0049                 I(i, i-3:i+2) = [12 -100 600 600 -100 12];
0050             end
0051             denom = 1024;
0052 
0053         case 8
0054             I(1, 1) = 1; I(end, end) = 1;
0055             A = [6435/32768 6435/4096 -15015/8192  9009/4096 -32175/16384  5005/4096 -4095/8192  495/4096 -429/32768; ...
0056                  -429/32768 1287/4096   9009/8192 -3003/4096   9009/16384 -1287/4096  1001/8192 -117/4096   99/32768; ...
0057                    99/32768 -165/4096   3465/8192  3465/4096  -5775/16384   693/4096  -495/8192   55/4096  -45/32768];
0058             I(2:4, 1:9) = A;
0059             I(m-1:m+1, m-7:end) = flipud(fliplr(A));%#ok
0060             for i = 5:m-2
0061                 I(i, i-4:i+3) = (1/2048).*[-5 49 -245 1225 1225 -245 49 -5];
0062             end
0063             denom = 1;   
0064 
0065     end
0066     I = (1/denom).*I;
0067 end

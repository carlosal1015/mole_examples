0017 function [Nx, Ny] = nodal2DCurv(k, X, Y)
0018     % Get the determinant of the jacobian and the metrics
0019     [J, Xe, Xn, Ye, Yn] = jacobian2D(k, X, Y);
0020     
0021     % Dimensions of nodal grid
0022     [n, m] = size(X);
0023     
0024     len = n*m;
0025     
0026     % Convert metrics to diagonal matrices
0027     J = spdiags(1./J, 0, len, len);
0028     Xe = spdiags(Xe, 0, len, len);
0029     Xn = spdiags(Xn, 0, len, len);
0030     Ye = spdiags(Ye, 0, len, len);
0031     Yn = spdiags(Yn, 0, len, len);
0032     
0033     % Construct 2D uniform nodal operator
0034     N = nodal2D(k, m, 1, n, 1); % N is tall and skinny
0035     Ne = N(1:len, :);
0036     Nn = N(len+1:end, :);
0037     
0038     % Apply transformation
0039     Nx = J*(Yn*Ne-Ye*Nn);
0040     Ny = J*(-Xn*Ne+Xe*Nn);
0041     
0042     % [J*Yn -J*Ye; J*-Xn J*Xe]*[Ne; Nn]
0043 end

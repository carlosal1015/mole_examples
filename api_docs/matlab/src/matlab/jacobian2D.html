0017 function [J, Xe, Xn, Ye, Yn] = jacobian2D(k, X, Y)
0018 % Returns:
0019 %                J : Determinant of the Jacobian (XeYn - XnYe)
0020 %               Xe : dx/de metric
0021 %               Xn : dx/dn metric
0022 %               Ye : dy/de metric
0023 %               Yn : dy/dn metric
0024 %
0025 % Parameters:
0026 %                k : Order of accuracy
0027 %                X : x-coordinates (physical) of meshgrid
0028 %                Y : y-coordinates (physical) of meshgrid
0029     
0030     [n, m] = size(X);
0031     
0032     X = reshape(X', [], 1);
0033     Y = reshape(Y', [], 1);
0034     
0035     N = nodal2D(k, m, 1, n, 1);
0036     
0037     X = N*X;
0038     Y = N*Y;
0039     
0040     mn = m*n;
0041     
0042     Xe = X(1:mn);
0043     Xn = X(mn+1:end);
0044     Ye = Y(1:mn);
0045     Yn = Y(mn+1:end);
0046     
0047     J = Xe.*Yn-Xn.*Ye;
0048 end

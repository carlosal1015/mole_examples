0017 function BC = mixedBC(k, m, dx, left, coeffs_left, right, coeffs_right)
0018 % Constructs a 1D mimetic mixed boundary conditions operator
0019 %
0020 % Parameters:
0021 %    k            : Order of accuracy
0022 %    m            : Number of cells
0023 %    dx           : Step size
0024 %    left         : Type of boundary condition at the left boundary ('Dirichlet', 'Neumann', 'Robin')
0025 %    coeffs_left  : Coefficients for the left boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0026 %    right        : Type of boundary condition at the right boundary ('Dirichlet', 'Neumann', 'Robin')
0027 %    coeffs_right : Coefficients for the right boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0028 
0029     A = sparse(m+2, m+2);
0030     B = sparse(m+2, m+1);
0031 
0032     switch left
0033         case 'Dirichlet'
0034             A(1, 1) = coeffs_left;
0035         case 'Neumann'
0036             B(1, 1) = -coeffs_left;
0037         case 'Robin'
0038             A(1, 1) = coeffs_left(1);
0039             B(1, 1) = -coeffs_left(2);
0040         otherwise
0041             error('Unknown boundary condition type');
0042     end
0043     
0044     switch right
0045         case 'Dirichlet'
0046             A(end, end) = coeffs_right;
0047         case 'Neumann'
0048             B(end, end) = coeffs_right;
0049         case 'Robin'
0050             A(end, end) = coeffs_right(1);
0051             B(end, end) = coeffs_right(2);
0052         otherwise
0053             error('Unknown boundary condition type');
0054     end
0055     
0056     G = grad(k, m, dx);
0057 
0058     BC = A + B*G;
0059 end

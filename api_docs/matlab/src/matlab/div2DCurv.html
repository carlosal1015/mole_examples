0017 function D = div2DCurv(k, X, Y)
0018 % Returns a 2D curvilinear mimetic divergence
0019 
0020     % Get the determinant of the jacobian and the metrics
0021     [J, Xe, Xn, Ye, Yn] = jacobian2D(k, X, Y);
0022     
0023     % Dimensions of nodal grid
0024     [n, m] = size(X);
0025     
0026     % Make them surfaces so they can be interpolated
0027     J = reshape(J, m, n)';
0028     Xe = reshape(Xe, m, n)';
0029     Xn = reshape(Xn, m, n)';
0030     Ye = reshape(Ye, m, n)';
0031     Yn = reshape(Yn, m, n)';
0032     
0033     % Logical grids
0034     [Xl, Yl] = meshgrid(1:m, 1:n);
0035     % Staggered logical grid
0036     [Xs, Ys] = meshgrid([1 1.5 : 1 : m-0.5 m], [1 1.5 : 1 : n-0.5 n]);
0037     
0038     % Interpolate the metrics on the logical grid for positions (Xs, Ys)
0039     J = interp2(Xl, Yl, J, Xs, Ys);
0040     Xe = interp2(Xl, Yl, Xe, Xs, Ys);
0041     Xn = interp2(Xl, Yl, Xn, Xs, Ys);
0042     Ye = interp2(Xl, Yl, Ye, Xs, Ys);
0043     Yn = interp2(Xl, Yl, Yn, Xs, Ys);
0044     
0045     % Convert metrics to diagonal matrices so they can be multiplied by the
0046     % logical operators
0047     J = spdiags(1./reshape(J', [], 1), 0, numel(J), numel(J));
0048     Xe = spdiags(reshape(Xe', [], 1), 0, numel(Xe), numel(Xe));
0049     Xn = spdiags(reshape(Xn', [], 1), 0, numel(Xn), numel(Xn));
0050     Ye = spdiags(reshape(Ye', [], 1), 0, numel(Ye), numel(Ye));
0051     Yn = spdiags(reshape(Yn', [], 1), 0, numel(Yn), numel(Yn));
0052     
0053     % Construct 2D uniform mimetic divergence operator (d/de, d/dn)
0054     D = div2D(k, m-1, 1, n-1, 1);
0055     De = D(:, 1:m*(n-1));
0056     Dn = D(:, m*(n-1)+1:end);
0057     
0058     % Apply transformation
0059     Dx = J*(Yn*De-Ye*DI2(m-1, n-1, 'Dn'));
0060     Dy = J*(-Xn*DI2(m-1, n-1, 'De')+Xe*Dn);
0061     
0062     % Final 2D curvilinear mimetic divergence operator (d/dx, d/dy)
0063     D = [Dx Dy];
0064 end

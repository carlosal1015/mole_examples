0017 function G = grad3DCurv(k, X, Y, Z)
0018 % Returns a 3D curvilinear mimetic gradient
0019 
0020     % Get the determinant of the jacobian and the metrics
0021     [J, Xe, Xn, Xc, Ye, Yn, Yc, Ze, Zn, Zc] = jacobian3D(k, X, Y, Z);
0022     
0023     % Dimensions of nodal grid
0024     [n, m, o] = size(X);
0025     
0026     % Make them volumes so they can be interpolated
0027     J = permute(reshape(J, m, n, o), [2, 1, 3]);
0028     A = permute(reshape(Yn.*Zc-Zn.*Yc, m, n, o), [2, 1, 3]);
0029     B = permute(reshape(Zn.*Xc-Xn.*Zc, m, n, o), [2, 1, 3]);
0030     C = permute(reshape(Xn.*Yc-Yn.*Xc, m, n, o), [2, 1, 3]);
0031     D = permute(reshape(Ze.*Yc-Ye.*Zc, m, n, o), [2, 1, 3]);
0032     E = permute(reshape(Xe.*Zc-Ze.*Xc, m, n, o), [2, 1, 3]);
0033     F = permute(reshape(Ye.*Xc-Xe.*Yc, m, n, o), [2, 1, 3]);
0034     G = permute(reshape(Ye.*Zn-Ze.*Yn, m, n, o), [2, 1, 3]);
0035     H = permute(reshape(Ze.*Xn-Xe.*Zn, m, n, o), [2, 1, 3]);
0036     I = permute(reshape(Xe.*Yn-Ye.*Xn, m, n, o), [2, 1, 3]);
0037     
0038     % Logical grids
0039     [Xl, Yl, Zl] = meshgrid(1:m, 1:n, 1:o);
0040     
0041     % Interpolate the metrics on the logical grid for positions u, v and w
0042     Xl_ = (Xl(1:end-1, :, :)+Xl(2:end, :, :))/2;
0043     Xl_ = (Xl_(:, :, 1:end-1)+Xl_(:, :, 2:end))/2;
0044     Yl_ = (Yl(1:end-1, :, :)+Yl(2:end, :, :))/2;
0045     Yl_ = (Yl_(:, :, 1:end-1)+Yl_(:, :, 2:end))/2;
0046     Zl_ = (Zl(1:end-1, :, :)+Zl(2:end, :, :))/2;
0047     Zl_ = (Zl_(:, :, 1:end-1)+Zl_(:, :, 2:end))/2;

0049     A = interp3(Xl, Yl, Zl, A, Xl_, Yl_, Zl_);
0050     D = interp3(Xl, Yl, Zl, D, Xl_, Yl_, Zl_);
0051     G = interp3(Xl, Yl, Zl, G, Xl_, Yl_, Zl_);
0052     
0053     Xl_ = (Xl(:, 1:end-1, :)+Xl(:, 2:end, :))/2;
0054     Xl_ = (Xl_(:, :, 1:end-1)+Xl_(:, :, 2:end))/2;
0055     Yl_ = (Yl(:, 1:end-1, :)+Yl(:, 2:end, :))/2;
0056     Yl_ = (Yl_(:, :, 1:end-1)+Yl_(:, :, 2:end))/2;
0057     Zl_ = (Zl(:, 1:end-1, :)+Zl(:, 2:end, :))/2;
0058     Zl_ = (Zl_(:, :, 1:end-1)+Zl_(:, :, 2:end))/2;
0059     Jv = interp3(Xl, Yl, Zl, J, Xl_, Yl_, Zl_);
0060     B = interp3(Xl, Yl, Zl, B, Xl_, Yl_, Zl_);
0061     E = interp3(Xl, Yl, Zl, E, Xl_, Yl_, Zl_);
0062     H = interp3(Xl, Yl, Zl, H, Xl_, Yl_, Zl_);
0063     
0064     Xl_ = (Xl(1:end-1, :, :)+Xl(2:end, :, :))/2;
0065     Xl_ = (Xl_(:, 1:end-1, :)+Xl_(:, 2:end, :))/2;
0066     Yl_ = (Yl(1:end-1, :, :)+Yl(2:end, :, :))/2;
0067     Yl_ = (Yl_(:, 1:end-1, :)+Yl_(:, 2:end, :))/2;
0068     Zl_ = (Zl(1:end-1, :, :)+Zl(2:end, :, :))/2;
0069     Zl_ = (Zl_(:, 1:end-1, :)+Zl_(:, 2:end, :))/2;
0070     Jw = interp3(Xl, Yl, Zl, J, Xl_, Yl_, Zl_);
0071     C = interp3(Xl, Yl, Zl, C, Xl_, Yl_, Zl_);
0072     F = interp3(Xl, Yl, Zl, F, Xl_, Yl_, Zl_);
0073     I = interp3(Xl, Yl, Zl, I, Xl_, Yl_, Zl_);
0074     
0075     % Convert metrics to diagonal matrices so they can be multiplied by the
0076     % logical operators
0077     Ju = spdiags(1./reshape(permute(Ju, [2, 1, 3]), [], 1), 0, numel(Ju), numel(Ju));
0078     Jv = spdiags(1./reshape(permute(Jv, [2, 1, 3]), [], 1), 0, numel(Jv), numel(Jv));
0079     Jw = spdiags(1./reshape(permute(Jw, [2, 1, 3]), [], 1), 0, numel(Jw), numel(Jw));

0082     C = spdiags(reshape(permute(C, [2, 1, 3]), [], 1), 0, numel(C), numel(C));
0083     D = spdiags(reshape(permute(D, [2, 1, 3]), [], 1), 0, numel(D), numel(D));
0084     E = spdiags(reshape(permute(E, [2, 1, 3]), [], 1), 0, numel(E), numel(E));
0085     F = spdiags(reshape(permute(F, [2, 1, 3]), [], 1), 0, numel(F), numel(F));
0086     G = spdiags(reshape(permute(G, [2, 1, 3]), [], 1), 0, numel(G), numel(G));
0087     H = spdiags(reshape(permute(H, [2, 1, 3]), [], 1), 0, numel(H), numel(H));
0088     I = spdiags(reshape(permute(I, [2, 1, 3]), [], 1), 0, numel(I), numel(I));
0089     
0090     % Construct 3D uniform mimetic gradient operator (d/de, d/dn, d/dc)
0091     Grad = grad3D(k, m-1, 1, n-1, 1, o-1, 1);
0092     Ge = Grad(1:m*(n-1)*(o-1), :);
0093     Gn = Grad(m*(n-1)*(o-1)+1:m*(n-1)*(o-1)+(m-1)*n*(o-1), :);
0094     Gc = Grad(m*(n-1)*(o-1)+(m-1)*n*(o-1)+1:end, :);
0095     
0096     % Apply transformation
0097     Gx = Ju*(A*Ge+D*GI13(Gn, m-1, n-1, o-1, 'Gn')+G*GI13(Gc, m-1, n-1, o-1, 'Gc'));
0098     Gy = Jv*(B*GI13(Ge, m-1, n-1, o-1, 'Ge')+E*Gn+H*GI13(Gc, m-1, n-1, o-1, 'Gcy'));
0099     Gz = Jw*(C*GI13(Ge, m-1, n-1, o-1, 'Gee')+F*GI13(Gn, m-1, n-1, o-1, 'Gnn')+I*Gc);
0100     
0101     % Final 3D curvilinear mimetic gradient operator (d/dx, d/dy, d/dz)
0102     G = [Gx; Gy; Gz];

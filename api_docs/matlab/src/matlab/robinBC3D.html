0017 function BC = robinBC3D(k, m, dx, n, dy, o, dz, a, b)
0018 % Returns a three-dimensional mimetic boundary operator that
0019 % imposes a boundary condition of Robin's type
0020 %
0021 % Parameters:
0022 %                k : Order of accuracy
0023 %                m : Number of cells along x-axis
0024 %               dx : Step size along x-axis
0025 %                n : Number of cells along y-axis
0026 %               dy : Step size along y-axis
0027 %                o : Number of cells along z-axis
0028 %               dz : Step size along z-axis
0029 %                a : Dirichlet Coefficient
0030 %                b : Neumann Coefficient
0031 
0032     % 1-D boundary operators
0033     Bm = robinBC(k, m, dx, a, b);
0034     Bn = robinBC(k, n, dy, a, b);
0035     Bo = robinBC(k, o, dz, a, b);
0036     
0037     Im = speye(m+2);
0038     
0039     In = speye(n+2);
0040     
0041     Io = speye(o+2);
0042     Io(1, 1) = 0;
0043     Io(end, end) = 0;
0044     
0045     In2 = In;
0046     In2(1, 1) = 0;
0047     In2(end, end) = 0;
0048     
0049     BC1 = kron(kron(Io, In2), Bm);
0050     BC2 = kron(kron(Io, Bn), Im);
0051     BC3 = kron(kron(Bo, In), Im);
0052     
0053     BC = BC1 + BC2 + BC3;
0054 end

0017 function [t, y] = rk4(func, tspan, dt, y0)
0018 % Explicit Runge-Kutta 4th-order method
0019 %
0020 % Returns: t (evaluation points) and y (solutions) of the specified ODE
0021 %
0022 % Parameters:
0023 %                func : Function handler
0024 %               tspan : [t0 tf]
0025 %                  dt : Step size
0026 %                  y0 : Initial conditions
0027 
0028     t = tspan(1) : dt : tspan(2);
0029     y = zeros(length(y0), length(t));
0030     y(:, 1) = y0;
0031     
0032     for i = 1 : length(t) - 1
0033         k1 = func(t(i),        y(:, i));
0034         k2 = func(t(i) + dt/2, y(:, i) + dt/2*k1);
0035         k3 = func(t(i) + dt/2, y(:, i) + dt/2*k2);
0036         k4 = func(t(i) + dt,   y(:, i) + dt*k3);
0037         
0038         y(:, i + 1) = y(:, i) + dt/6*(k1 + 2*k2 + 2*k3 + k4);
0039     end
0040 end

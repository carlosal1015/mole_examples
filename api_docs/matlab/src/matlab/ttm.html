0018 function [X, Y] = ttm(grid_name, m, n, iters, plot_grid)
0019 % Returns X and Y which are both m by n matrices that contains the physical
0020 % coordinates
0021 %
0022 % Parameters:
0023 %        grid_name : String with the name of the grid folder
0024 %                m : Number of nodes along the horizontal axis
0025 %                n : Number of nodes along the vertical axis
0026 %        plot_grid : If defined -> grid will be plotted
0027     
0028     assert(m > 4 && n > 4, 'm and n must be greater than 4')
0029     
0030     addpath(['grids/' grid_name])
0031     
0032     % Error tolerance for iterative method
0033     tol = 10^-6;
0034     
0035     % Preallocation
0036     X = zeros(m, n);
0037     Y = zeros(m, n);
0038     alpha = zeros(m, n);
0039     beta = zeros(m, n);
0040     gamma = zeros(m, n);
0041     
0042     % BCs
0043     for i = 1 : m
0044         xi = (i-1)/(m-1);
0045         XY = top(xi);
0046         X(i, end) = XY(1);
0047         Y(i, end) = XY(2);
0048         XY = bottom(xi);
0049         X(i, 1) = XY(1);
0050         Y(i, 1) = XY(2);
0051     end
0052     for j = 1 : n
0053         eta = (j-1)/(n-1);
0054         XY = left(eta);
0055         X(1, j) = XY(1);
0056         Y(1, j) = XY(2);
0057         XY = right(eta);
0058         X(end, j) = XY(1);
0059         Y(end, j) = XY(2);
0060     end
0061     
0062     newX = X;
0063     newY = Y;
0064     
0065     errX = zeros(1, iters);
0066     errY = zeros(1, iters);
0067     
0068     % SOR
0069     for t = 1 : iters
0070         i = 2 : m-1;
0071         j = 2 : n-1;
0072         
0073         alpha(i, j) = 0.25*((X(i, j+1)-X(i, j-1)).^2+(Y(i, j+1)-Y(i, j-1)).^2);
0074         beta(i, j) = 0.0625*((X(i+1, j)-X(i-1, j)).*(X(i, j+1)-X(i, j-1))+(Y(i+1, j)...
0075            -Y(i-1, j)).*(Y(i, j+1)-Y(i, j-1)));
0076         gamma(i, j) = 0.25*((X(i+1, j)-X(i-1, j)).^2+(Y(i+1, j)-Y(i-1, j)).^2);
0077         
0078         newX(i, j) = ((-0.5)./(alpha(i, j)+gamma(i, j)+1e-10)).*(2*beta(i, j)...
0079             .*(X(i+1, j+1)-X(i-1, j+1)-X(i+1, j-1)+X(i-1, j-1))-alpha(i, j)...
0080             .*(X(i+1, j)+X(i-1, j))-gamma(i, j).*(X(i, j+1)+X(i, j-1)));
0081         newY(i, j) = ((-0.5)./(alpha(i, j)+gamma(i, j)+1e-10)).*(2*beta(i, j)...
0082             .*(Y(i+1, j+1)-Y(i-1, j+1)-Y(i+1, j-1)+Y(i-1, j-1))-alpha(i, j)...
0083             .*(Y(i+1, j)+Y(i-1, j))-gamma(i, j).*(Y(i, j+1)+Y(i, j-1)));
0084         
0085         errX(1, t) = max(max(abs(newX-X)));
0086         errY(1, t) = max(max(abs(newY-Y)));
0087         
0088         % Update
0089         X = newX;
0090         Y = newY;
0091         
0092         if errX(t) < tol && errY(t) < tol
0093             break
0094         end
0095     end
0096     
0097     if plot_grid
0098         figure
0099         mesh(X, Y, zeros(m, n), 'Marker', '.', 'MarkerSize', 10, 'EdgeColor', 'b')
0100         title(['Physical grid. m = ' num2str(m) ', n = ' num2str(n)])
0101         set(gcf, 'color', 'w')
0102         axis equal
0103         axis off
0104         view([0 90])
0105     end
0106 end

0017 function D = div3DCurv(k, X, Y, Z)
0018 % Returns a 3D curvilinear mimetic divergence
0019 
0020     % Get the determinant of the jacobian and the metrics
0021     [J, Xe, Xn, Xc, Ye, Yn, Yc, Ze, Zn, Zc] = jacobian3D(k, X, Y, Z);
0022     
0023     % Dimensions of nodal grid
0024     [n, m, o] = size(X);
0025     
0026     % Make them volumes so they can be interpolated
0027     J = permute(reshape(J, m, n, o), [2, 1, 3]);
0028     A = permute(reshape(Yn.*Zc-Zn.*Yc, m, n, o), [2, 1, 3]);
0029     B = permute(reshape(Zn.*Xc-Xn.*Zc, m, n, o), [2, 1, 3]);
0030     C = permute(reshape(Xn.*Yc-Yn.*Xc, m, n, o), [2, 1, 3]);
0031     D = permute(reshape(Ze.*Yc-Ye.*Zc, m, n, o), [2, 1, 3]);
0032     E = permute(reshape(Xe.*Zc-Ze.*Xc, m, n, o), [2, 1, 3]);
0033     F = permute(reshape(Ye.*Xc-Xe.*Yc, m, n, o), [2, 1, 3]);
0034     G = permute(reshape(Ye.*Zn-Ze.*Yn, m, n, o), [2, 1, 3]);
0035     H = permute(reshape(Ze.*Xn-Xe.*Zn, m, n, o), [2, 1, 3]);
0036     I = permute(reshape(Xe.*Yn-Ye.*Xn, m, n, o), [2, 1, 3]);
0037     
0038     % Logical grids
0039     [Xl, Yl, Zl] = meshgrid(1:m, 1:n, 1:o);
0040     % Staggered logical grid
0041     [Xs, Ys, Zs] = meshgrid([1 1.5 : 1 : m-0.5 m], [1 1.5 : 1 : n-0.5 n], [1 1.5 : 1 : o-0.5 o]);
0042     
0043     % Interpolate the metrics on the logical grid for positions (Xs, Ys, Zs)
0044     J = interp3(Xl, Yl, Zl, J, Xs, Ys, Zs);
0045     A = interp3(Xl, Yl, Zl, A, Xs, Ys, Zs);
0046     B = interp3(Xl, Yl, Zl, B, Xs, Ys, Zs);
0047     C = interp3(Xl, Yl, Zl, C, Xs, Ys, Zs);
0048     D = interp3(Xl, Yl, Zl, D, Xs, Ys, Zs);
0049     E = interp3(Xl, Yl, Zl, E, Xs, Ys, Zs);
0050     F = interp3(Xl, Yl, Zl, F, Xs, Ys, Zs);
0051     G = interp3(Xl, Yl, Zl, G, Xs, Ys, Zs);
0052     H = interp3(Xl, Yl, Zl, H, Xs, Ys, Zs);
0053     I = interp3(Xl, Yl, Zl, I, Xs, Ys, Zs);
0054     
0055     % Convert metrics to diagonal matrices so they can be multiplied by the
0056     % logical operators
0057     J = spdiags(1./reshape(permute(J, [2, 1, 3]), [], 1), 0, numel(J), numel(J));
0058     A = spdiags(reshape(permute(A, [2, 1, 3]), [], 1), 0, numel(A), numel(A));
0059     B = spdiags(reshape(permute(B, [2, 1, 3]), [], 1), 0, numel(B), numel(B));
0060     C = spdiags(reshape(permute(C, [2, 1, 3]), [], 1), 0, numel(C), numel(C));
0061     D = spdiags(reshape(permute(D, [2, 1, 3]), [], 1), 0, numel(D), numel(D));
0062     E = spdiags(reshape(permute(E, [2, 1, 3]), [], 1), 0, numel(E), numel(E));
0063     F = spdiags(reshape(permute(F, [2, 1, 3]), [], 1), 0, numel(F), numel(F));
0064     G = spdiags(reshape(permute(G, [2, 1, 3]), [], 1), 0, numel(G), numel(G));
0065     H = spdiags(reshape(permute(H, [2, 1, 3]), [], 1), 0, numel(H), numel(H));
0066     I = spdiags(reshape(permute(I, [2, 1, 3]), [], 1), 0, numel(I), numel(I));
0067     
0068     % Construct 3D uniform mimetic divergence operator (d/de + d/dn + d/dc)
0069     Div = div3D(k, m-1, 1, n-1, 1, o-1, 1);
0070     De = Div(:, 1:m*(n-1)*(o-1));
0071     Dn = Div(:, m*(n-1)*(o-1)+1:m*(n-1)*(o-1)+(m-1)*n*(o-1));
0072     Dc = Div(:, m*(n-1)*(o-1)+(m-1)*n*(o-1)+1:end);
0073     
0074     % Apply transformation
0075     Dx = J*(A*De+D*DI3(m-1, n-1, o-1, 'Dn')+G*DI3(m-1, n-1, o-1, 'Dc'));
0076     Dy = J*(B*DI3(m-1, n-1, o-1, 'De')+E*Dn+H*DI3(m-1, n-1, o-1, 'Dcc'));
0077     Dz = J*(C*DI3(m-1, n-1, o-1, 'Dee')+F*DI3(m-1, n-1, o-1, 'Dnn')+I*Dc);
0078     
0079     % Final 3D curvilinear mimetic divergence operator (d/dx + d/dy + d/dz)
0080     D = [Dx Dy Dz];
0081 end

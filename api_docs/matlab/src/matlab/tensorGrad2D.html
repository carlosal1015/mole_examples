0017 function KG = tensorGrad2D(K, G)
0018 % Returns a two-dimensional flux operator
0019 %
0020 % Parameters:
0021 %                K : Tensor (e.g. diffusion tensor)
0022 %                G : 2D mimetic gradient operator
0023 
0024     rowsGx = find(G(:, 2), 1)-1;
0025     rowsGy = size(G, 1)-rowsGx;
0026     
0027     Gx = G(1:rowsGx, :);
0028     Gy = G(rowsGx+1:end, :);
0029     
0030     if size(K, 1) ~= 2
0031         Kxx = spdiags(cell2mat(K(1)), 0, rowsGx, rowsGx);
0032         Kyy = spdiags(cell2mat(K(2)), 0, rowsGy, rowsGy);
0033         %Kxy = spdiags(cell2mat(K(3)), 0, rowsGy, rowsGy);
0034         %Kyx = spdiags(cell2mat(K(4)), 0, rowsGx, rowsGx);
0035     else
0036         Kxx = K(1, 1);
0037         Kyy = K(2, 2);
0038         %Kxy = K(1, 2);
0039         %Kyx = K(2, 1);
0040     end
0041     
0042     % The following line and lines [17, 18, 22, 23] are commented
0043     % because Gx and Gy will have incompatible sizes unless m == n
0044     % OR if G is a nodal operator.
0045     %KG = [Kxx*Gx + Kxy*Gy; Kyx*Gx + Kyy*Gy];
0046     KG = [Kxx*Gx; Kyy*Gy];
0047 end

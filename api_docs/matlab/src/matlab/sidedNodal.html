0017 function S = sidedNodal(m, dx, type)
0018 % Returns a m+1 by m+1 one-dimensional sided approximation for uniformly
0019 % spaced data points. This function is handy for advective terms.
0020 %
0021 % Parameters:
0022 %                m : Number of cells
0023 %               dx : Step size
0024 %             type : 'backward', 'forward' or 'centered'
0025 
0026     switch type
0027         case 'backward'
0028             S = spdiags([-ones(m+1, 1) ones(m+1, 1)], [-1 0], m+1, m+1);
0029             S(1, end-1) = -1;
0030             S = S/dx;
0031         case 'forward'
0032             S = spdiags([-ones(m+1, 1) ones(m+1, 1)], [0 1], m+1, m+1);
0033             S(end, 2) = 1;
0034             S = S/dx;
0035         otherwise % 'centered'
0036             S = spdiags([-ones(m+1, 1) zeros(m+1, 1) ones(m+1, 1)], [-1 0 1], m+1, m+1);
0037             S(1, end-1) = -1;
0038             S(end, 2) = 1;
0039             S = S/(2*dx);
0040     end
0041 end

0017 function I = interpolFacesToCentersG3D(k, m, n, o)
0018 % 3D interpolation from faces to centers
0019 % centers logical coordinates [1,1.5:m-0.5,m]x[1,1.5:n-0.5,n]x[1,1.5:o-0.5,o]
0020 % m, n, o, are the number of cells in the logical x-, y-, z- axes
0021 
0022     cells = (o+2)*(n+2)*(m+2);
0023 
0024     Ix = interpolFacesToCentersG1D(k, m);
0025     Iy = interpolFacesToCentersG1D(k, n);
0026     Iz = interpolFacesToCentersG1D(k, o);
0027 
0028     Im = sparse(m + 2, m);
0029     In = sparse(n + 2, n);
0030     Io = sparse(o + 2, o);
0031 
0032     Im(2:(m+2)-1, :) = speye(m, m);
0033     In(2:(n+2)-1, :) = speye(n, n);
0034     Io(2:(o+2)-1, :) = speye(o, o);
0035 
0036     Sx = kron(kron(Io, In), Ix);
0037     Sy = kron(kron(Io, Iy), Im);
0038     Sz = kron(kron(Iz, In), Im);
0039 
0040     I = sparse(3*cells, o*n*(m+1)+o*(n+1)*m+(o+1)*n*m);
0041     
0042     I(1:cells, 1:o*n*(m+1)) = Sx; 
0043     I(cells+1:2*cells, o*n*(m+1)+1:o*n*(m+1)+o*(n+1)*m) = Sy;  
0044     I(2*cells+1:end, o*n*(m+1)+o*(n+1)*m+1:end) = Sz;
0045 end

0017 function I = GI1(M, m, n, type)
0018     if strcmp(type, 'Gn')
0019         I = speye(n);
0020         I1 = speye(m+1, m);
0021         I1(end, end) = 1;
0022         I = kron(I, I1);
0023         I = [I sparse(size(I, 1), m)];
0024         I = I*M;
0025     else
0026         I = speye(n+1, n);
0027         I(end, end) = 1;
0028         I1 = speye(m, m+1);
0029         I = kron(I, I1);
0030         I = I*M;
0031     end
0032 end

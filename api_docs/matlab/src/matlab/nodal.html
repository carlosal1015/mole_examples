0017 function N = nodal(k, m, dx)
0018 % Returns a m+1 by m+1 one-dimensional operator that approximates the
0019 % first-order derivatives on a uniform nodal grid
0020 %
0021 % Parameters:
0022 %                k : Order of accuracy
0023 %                m : Number of nodes
0024 %               dx : Step size
0025 
0026     m = m-1;
0027     n_rows = m+1;
0028     n_cols = n_rows;
0029 
0030     N = sparse(n_rows, n_cols);
0031 
0032     neighbors = zeros(1, k+1); % Bandwidth = k+1
0033     neighbors(1) = -k/2;
0034     len = k+1;
0035     j = 1;
0036 
0037     for i = 2 : len
0038         neighbors(i) = neighbors(i-1)+1;
0039     end
0040 
0041     % Create a k by k Vandermonde matrix based on the neighbors:
0042     A = vander(neighbors)';
0043 
0044     % First-order derivative
0045     b = zeros(len, 1);
0046     b(len-1) = 1;
0047 
0048     % Solve the linear system to get the coefficients
0049     coeffs = A\b;
0050 
0051     for i = k/2+1 : n_rows-k/2
0052         N(i, j:j+len-1) = coeffs;
0053         j = j+1;
0054     end
0055 
0056     p = k/2;
0057     q = k+1;
0058     A = sparse(p, q);
0059     for i = 1 : p % For each row of A
0060         neighbors = zeros(1, q); % k+1 points are used for the boundaries
0061         neighbors(1) = 1-i; % Shifting the stencil to the right
0062         neighbors(2) = neighbors(1)+1;
0063 
0064         for j = 3 : q
0065             neighbors(j) = neighbors(j-1)+1;
0066         end
0067 
0068         V = vander(neighbors)';
0069         b = zeros(q, 1);
0070         b(q-1) = 1;
0071         coeffs = V\b;
0072         A(i, 1:q) = coeffs;
0073     end
0074 
0075     % Insert A into N (upper-left corner of N)
0076     N(1:p, 1:q) = A;
0077 
0078     % Permutation matrices
0079     Pp = fliplr(speye(p));
0080     Pq = fliplr(speye(q));
0081     % Construct A' (lower-right corner of N)
0082     A = -Pp*A*Pq;
0083 
0084     % Insert A' into N
0085     N(n_rows-p+1:n_rows, n_cols-q+1:n_cols) = A;
0086 
0087     % Scale N
0088     N = 1/dx*N;
0089 end

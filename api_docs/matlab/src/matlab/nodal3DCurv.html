0017 function [Nx, Ny, Nz] = nodal3DCurv(k, X, Y, Z)
0018     % Get the determinant of the jacobian and the metrics
0019     [J, Xe, Xn, Xc, Ye, Yn, Yc, Ze, Zn, Zc] = jacobian3D(k, X, Y, Z);
0020     
0021     % Dimensions of nodal grid
0022     [n, m, o] = size(X);
0023     
0024     len = n*m*o;
0025     
0026     % Convert metrics to diagonal matrices
0027     J = spdiags(1./J, 0, numel(len), numel(len));
0028     A = spdiags(Yn.*Zc-Zn.*Yc, 0, len, len);
0029     B = spdiags(Zn.*Xc-Xn.*Zc, 0, len, len);
0030     C = spdiags(Xn.*Yc-Yn.*Xc, 0, len, len);
0031     D = spdiags(Ze.*Yc-Ye.*Zc, 0, len, len);
0032     E = spdiags(Xe.*Zc-Ze.*Xc, 0, len, len);
0033     F = spdiags(Ye.*Xc-Xe.*Yc, 0, len, len);
0034     G = spdiags(Ye.*Zn-Ze.*Yn, 0, len, len);
0035     H = spdiags(Ze.*Xn-Xe.*Zn, 0, len, len);
0036     I = spdiags(Xe.*Yn-Ye.*Xn, 0, len, len);
0037     
0038     % Construct 3D uniform nodal operator
0039     N = nodal3D(k, m, 1, n, 1, o, 1); % N is tall and skinny
0040     Ne = N(1:len, :);
0041     Nn = N(len+1:2*len, :);
0042     Nc = N(2*len+1:end, :);
0043     
0044     % Apply transformation
0045     Nx = J*(A*Ne+D*Nn+G*Nc);
0046     Ny = J*(B*Ne+E*Nn+H*Nc);
0047     Nz = J*(C*Ne+F*Nn+I*Nc);
0048     
0049     %  [J*A J*D J*G;
0050     %   J*B J*E J*H;
0051     %   J*C J*F J*I]*[Ne; Nn; Nc]
0052 end

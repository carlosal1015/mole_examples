0017 % https://en.wikipedia.org/wiki/Transfinite_interpolation
0018 function [X, Y] = tfi(grid_name, m, n, plot_grid)
0019 % Returns X and Y which are both m by n matrices that contains the physical
0020 % coordinates
0021 %
0022 % Parameters:
0023 %        grid_name : String with the name of the grid folder
0024 %                m : Number of nodes along the horizontal axis
0025 %                n : Number of nodes along the vertical axis
0026 %        plot_grid : If defined -> grid will be plotted
0027     
0028     assert(m > 4 && n > 4, 'm and n must be greater than 4')
0029     
0030     addpath(['grids/' grid_name])
0031     
0032     % Logical grid
0033     xi = linspace(0, 1, m);
0034     eta = linspace(0, 1, n);
0035     
0036     % Allocate space for physical grid
0037     X = zeros(m, n);
0038     Y = zeros(m, n);
0039     
0040     for i = 1 : m
0041         u = xi(i);
0042         for j = 1 : n
0043             v = eta(j);
0044             % Transfinite interpolation
0045             XY = (1-v)*bottom(u)+v*top(u)+(1-u)*left(v)+u*right(v)-...
0046                 (u*v*top(1)+u*(1-v)*bottom(1)+v*(1-u)*top(0)+(1-u)*(1-v)*bottom(0));
0047             X(i, j) = XY(1);
0048             Y(i, j) = XY(2);
0049         end
0050     end
0051     
0052     if plot_grid
0053         figure
0054         mesh(X, Y, zeros(m, n), 'Marker', '.', 'MarkerSize', 10, 'EdgeColor', 'b')
0055         title(['Physical grid. m = ' num2str(m) ', n = ' num2str(n)])
0056         set(gcf, 'color', 'w')
0057         axis equal
0058         axis off
0059         view([0 90])
0060     end
0061 end

0017 function BC = robinBC2D(k, m, dx, n, dy, a, b)
0018 % Returns a two-dimensional mimetic boundary operator that
0019 % imposes a boundary condition of Robin's type
0020 %
0021 % Parameters:
0022 %                k : Order of accuracy
0023 %                m : Number of cells along x-axis
0024 %               dx : Step size along x-axis
0025 %                n : Number of cells along y-axis
0026 %               dy : Step size along y-axis
0027 %                a : Dirichlet Coefficient
0028 %                b : Neumann Coefficient
0029 
0030     % 1-D boundary operator
0031     Bm = robinBC(k, m, dx, a, b);
0032     Bn = robinBC(k, n, dy, a, b);
0033     
0034     Im = speye(m+2);
0035     
0036     In = speye(n+2);
0037     In(1, 1) = 0;
0038     In(end, end) = 0;
0039     
0040     BC1 = kron(In, Bm);
0041     BC2 = kron(Bn, Im);
0042     
0043     BC = BC1 + BC2;
0044 end

0017 function I = interpolCentersToFacesD1D(k, m)
0018 % 1D interpolation from centers to faces.
0019 % logical centers are [1 1.5 2.5 ... m-1.5 m-0.5 m]
0020 % m is the number of cells in the logic x-axis
0021 
0022     I = sparse(m+1, m+2);
0023 
0024     switch k
0025         case 2
0026             I(1, 1) = 2; I(end, end) = 2;
0027             for i = 2:m
0028                I(i, i:i+1) = [1 1];
0029             end
0030             denom = 2;
0031 
0032         case 4
0033             I(1, 1) = 112; I(end, end) = 112;
0034             A = [-16 70 70 -14 2];
0035             I(2, 1:5) = A;
0036             I(m, m-2:end) = fliplr(A);
0037             for i = 3:m-1
0038                I(i, i-1:i+2) = [-7 63 63 -7];
0039             end
0040             denom = 112;
0041 
0042         case 6
0043             I(1, 1) = 8448; I(end, end) = 8448;
0044             A = [-768 4158 6930 -2772  1188 -330  42; ...
0045                   256 -924 4620  5544 -1320  308 -36];
0046             I(2:3, 1:7) = A;
0047             I(m-1:m, m-4:end) = flipud(fliplr(A));%#ok
0048             for i = 4:m-2
0049                 I(i, i-2:i+3) = [99 -825 4950 4950 -825 99];
0050             end
0051             denom = 8448;
0052 
0053         case 8
0054             I(1, 1) = 1; I(end, end) = 1;
0055             A = [-1/15  429/1024 1001/1024 -3003/5120  429/1024 -715/3072  91/1024   -21/1024  11/5120; ...
0056                   1/65  -33/512   231/512   2079/2560 -165/512    77/512  -27/512     77/6656  -3/2560; ...
0057                  -1/143  27/1024 -105/1024   567/1024  675/1024 -175/1024 567/11264 -135/13312  1/1024];
0058             I(2:4, 1:9) = A;

0062             end
0063             denom = 1;   
0064 
0065     end
0066     I = (1/denom).*I;

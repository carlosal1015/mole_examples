0017 function G = grad2DCurv(k, X, Y)
0018 % Returns a 2D curvilinear mimetic gradient
0019 
0020     % Get the determinant of the jacobian and the metrics
0021     [J, Xe, Xn, Ye, Yn] = jacobian2D(k, X, Y);
0022     
0023     % Dimensions of nodal grid
0024     [n, m] = size(X);
0025     
0026     % Make them surfaces so they can be interpolated
0027     J = reshape(J, m, n)';
0028     Xe = reshape(Xe, m, n)';
0029     Xn = reshape(Xn, m, n)';
0030     Ye = reshape(Ye, m, n)';
0031     Yn = reshape(Yn, m, n)';
0032     
0033     % Logical grids
0034     [Xl, Yl] = meshgrid(1:m, 1:n);
0035     
0036     % Interpolate the metrics on the logical grid for positions u and v
0037     Ju = interp2(Xl, Yl, J, (Xl(1:end-1, :)+Xl(2:end, :))/2,...
0038                                             (Yl(1:end-1, :)+Yl(2:end, :))/2);
0039     Jv = interp2(Xl, Yl, J, (Xl(:, 1:end-1)+Xl(:, 2:end))/2,...
0040                                             (Yl(:, 1:end-1)+Yl(:, 2:end))/2);
0041     Xev = interp2(Xl, Yl, Xe, (Xl(:, 1:end-1)+Xl(:, 2:end))/2,...
0042                                             (Yl(:, 1:end-1)+Yl(:, 2:end))/2);
0043     Xnv = interp2(Xl, Yl, Xn, (Xl(:, 1:end-1)+Xl(:, 2:end))/2,...
0044                                             (Yl(:, 1:end-1)+Yl(:, 2:end))/2);
0045     Yeu = interp2(Xl, Yl, Ye, (Xl(1:end-1, :)+Xl(2:end, :))/2,...
0046                                             (Yl(1:end-1, :)+Yl(2:end, :))/2);
0047     Ynu = interp2(Xl, Yl, Yn, (Xl(1:end-1, :)+Xl(2:end, :))/2,...
0048                                             (Yl(1:end-1, :)+Yl(2:end, :))/2);
0049     
0050     % Convert metrics to diagonal matrices so they can be multiplied by the
0051     % logical operators
0052     Ju = spdiags(1./reshape(Ju', [], 1), 0, numel(Ju), numel(Ju));
0053     Jv = spdiags(1./reshape(Jv', [], 1), 0, numel(Jv), numel(Jv));
0054     Xev = spdiags(reshape(Xev', [], 1), 0, numel(Xev), numel(Xev));
0055     Xnv = spdiags(reshape(Xnv', [], 1), 0, numel(Xnv), numel(Xnv));
0056     Yeu = spdiags(reshape(Yeu', [], 1), 0, numel(Yeu), numel(Yeu));
0057     Ynu = spdiags(reshape(Ynu', [], 1), 0, numel(Ynu), numel(Ynu));
0058     
0059     % Construct 2D uniform mimetic gradient operator (d/de, d/dn)
0060     G = grad2D(k, m-1, 1, n-1, 1);
0061     Ge = G(1:m*(n-1), :);
0062     Gn = G(m*(n-1)+1:end, :);
0063     
0064     % Apply transformation
0065     Gx = Ju*(Ynu*Ge-Yeu*GI2(Gn, m-1, n-1, 'Gn'));
0066     Gy = Jv*(-Xnv*GI2(Ge, m-1, n-1, 'Ge')+Xev*Gn);
0067     
0068     % Final 2D curvilinear mimetic gradient operator (d/dx, d/dy)
0069     G = [Gx; Gy];
0070 end

0017 function N = nodal2D(k, m, dx, n, dy)
0018 % Returns a two-dimensional operator that approximates the first-order
0019 % derivatives on a uniform nodal grid
0020 %
0021 % Parameters:
0022 %                k : Order of accuracy
0023 %                m : Number of nodes along x-axis
0024 %               dx : Step size along x-axis
0025 %                n : Number of nodes along y-axis
0026 %               dy : Step size along y-axis
0027     
0028     Nx = nodal(k, m, dx);
0029     Ny = nodal(k, n, dy);
0030     
0031     Im = speye(m, m);
0032     In = speye(n, n);
0033     
0034     N = [kron(In, Nx); kron(Ny, Im)];
0035 end

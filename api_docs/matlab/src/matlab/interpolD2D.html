0017 function I = interpolD2D(m, n, c1, c2)
0018 % Returns a two-dimensional interpolator of 2nd-order
0019 %                m : Number of cells along x-axis
0020 %                n : Number of cells along y-axis
0021 %               c1 : Left interpolation coeff.
0022 %               c2 : Bottom interpolation coeff.
0023 
0024     Ix = interpolD(m, c1);
0025     Iy = interpolD(n, c2);
0026     
0027     Im = sparse(m + 2, m);
0028     In = sparse(n + 2, n);
0029     
0030     Im(2:(m+2)-1, :) = speye(m, m);
0031     In(2:(n+2)-1, :) = speye(n, n);
0032     
0033     Sx = kron(In, Ix);
0034     Sy = kron(Iy, Im);
0035     
0036     I = [Sx Sy];
0037 end

0017 function BC = mixedBC2D(k, m, dx, n, dy, left, coeffs_left, right, coeffs_right, bottom, coeffs_bottom, top, coeffs_top)
0018 % Constructs a 2D mimetic mixed boundary conditions operator
0019 %
0020 % Parameters:
0021 %    k             : Order of accuracy
0022 %    m             : Number of cells in x-direction
0023 %    dx            : Step size in x-direction
0024 %    n             : Number of cells in y-direction
0025 %    dy            : Step size in y-direction
0026 %    left          : Type of boundary condition at the left boundary ('Dirichlet', 'Neumann', 'Robin')
0027 %    coeffs_left   : Coefficients for the left boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0028 %    right         : Type of boundary condition at the right boundary ('Dirichlet', 'Neumann', 'Robin')
0029 %    coeffs_right  : Coefficients for the right boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0030 %    bottom        : Type of boundary condition at the bottom boundary ('Dirichlet', 'Neumann', 'Robin')
0031 %    coeffs_bottom : Coefficients for the bottom boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0032 %    top           : Type of boundary condition at the top boundary ('Dirichlet', 'Neumann', 'Robin')
0033 %    coeffs_top    : Coefficients for the top boundary condition (a, b for Robin, otherwise coeff. for Dirichlet or Neumann)
0034 
0035     % 1-D boundary operators
0036     Bm = mixedBC(k, m, dx, left, coeffs_left, right, coeffs_right);
0037     Bn = mixedBC(k, n, dy, bottom, coeffs_bottom, top, coeffs_top);
0038     
0039     Im = speye(m+2);
0040     
0041     In = speye(n+2);
0042     In(1, 1) = 0;
0043     In(end, end) = 0;
0044     
0045     BC1 = kron(In, Bm);
0046     BC2 = kron(Bn, Im);
0047     
0048     BC = BC1 + BC2;
0049 end
